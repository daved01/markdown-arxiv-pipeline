% --- CSL REFERENCES SETUP ---
\newlength{\cslhangindent}
\setlength{\cslhangindent}{1.5em}
\newlength{\csllabelwidth}
\setlength{\csllabelwidth}{3em}

\newenvironment{CSLReferences}[2] 
 {%
  \begin{list}{}{%
   \setlength{\itemindent}{0pt}
   \setlength{\leftmargin}{0pt}
   \setlength{\parsep}{0pt}
   \setlength{\labelwidth}{0pt}
   \setlength{\labelsep}{0pt}
   \setlength{\itemsep}{#2\baselineskip}
   % Suppress the default list label
   \renewcommand{\makelabel}[1]{}
  }%
 }
 {\end{list}}

\newcommand{\CSLBlock}[1]{#1\hfill\break}

% \strut forces the height of the line to be consistent.
% \phantomsection creates the link target.
\newcommand{\CSLLeftMargin}[1]{\parbox[t]{\csllabelwidth}{\strut\phantomsection#1}}

% Subtract 1em from the width to prevent line-wrapping issues.
\newcommand{\CSLRightInline}[1]{\parbox[t]{\linewidth - \csllabelwidth - 1em}{\strut#1}\vspace{0.5em}} 
\newcommand{\CSLIndent}[1]{\hspace{\cslhangindent}#1}
